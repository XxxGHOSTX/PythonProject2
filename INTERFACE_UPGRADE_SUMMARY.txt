function dragWindow(e) {
    if (!isDragging) return;
    const window_el = document.getElementById('floating-window');
    window_el.style.left = (e.clientX - dragOffsetX) + 'px';
    window_el.style.top = (e.clientY - dragOffsetY) + 'px';
}
